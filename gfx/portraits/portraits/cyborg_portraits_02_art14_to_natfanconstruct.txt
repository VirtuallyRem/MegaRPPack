##############################################################
###	PORTRAIT GROUPS SETTINGS
###
###	This file configures when to display the cyborg version 
### or the organic version of a single species
##############################################################


portrait_groups = {
	art14 = {
		default = art14
		game_setup = { #will run with a limited country scope. species and government is set but the country does not actually exist
			add = {
				trigger = {
					species = { NOT = { has_trait = trait_cybernetic } }
				}
				portraits = {
					art14
				}
			}
			add = {
				trigger = {
					species = { has_trait = trait_cybernetic } 
				}
				portraits = {
					natfanconstruct
				}
			}

		}		
		
		#species scope
		species = { #generic portrait for a species
			add = {
				trigger = {
					NOT = { has_trait = trait_cybernetic }
				}
				portraits = {
					art14
				}
			}
			add = {
				trigger = {
					has_trait = trait_cybernetic 
				}
				portraits = {
					natfanconstruct
				}
			}
		}
		
		#pop scope
		pop = { #for a specific pop
			add = {
				trigger = {
					NOT = { has_trait = trait_cybernetic }
				}
				portraits = {
					art14
				}
			}
			add = {
				trigger = {
					has_trait = trait_cybernetic
				}
				portraits = {
					natfanconstruct
				}
			}
		}
		
		#leader scope
		leader = { #scientists, generals, admirals, governor
			add = {
				trigger = {
					NOR = {
						has_trait = leader_trait_admiral_cyborg
						has_trait = leader_trait_general_cyborg
						has_trait = leader_trait_governor_cyborg
						has_trait = leader_trait_ruler_cyborg
						has_trait = leader_trait_scientist_cyborg
					}
				}
				portraits = {
					art14
				}
			}
			add = {
				trigger = {
					OR = {
						has_trait = leader_trait_admiral_cyborg
						has_trait = leader_trait_general_cyborg
						has_trait = leader_trait_governor_cyborg
						has_trait = leader_trait_ruler_cyborg
						has_trait = leader_trait_scientist_cyborg
					}
				}
				portraits = {
					natfanconstruct
				}
			}
		}

			
		#leader scope 
		ruler = {
			add = {
				trigger = {
					NOR = {
						has_trait = leader_trait_admiral_cyborg

					}
				}

					natfanconstruct

			}
